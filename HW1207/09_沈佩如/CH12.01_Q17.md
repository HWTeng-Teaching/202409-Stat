$$\sum x = (-2)+(-1) + 0 + 1 + 2 = 0 $$

$$\sum y = 1 + 1 + 3 + 5 + 5 = 15 $$

$$\sum {x^2} = (-2)^2 +(-1)^2 + 0^2 + 1^2 + 2^2 = 10 $$

$$\sum {y^2} = 1^2 + 1^2 + 3^2 + 5^2 + 5^2 = 61 $$

$$\sum xy = (-2)(1) + (-1)(1) + (0)(3) + (1)(5) + (2)(5) = 12 $$

$$\bar{x} = \frac{(-2)+(-1)+0+1+2}{5} = 0 $$
